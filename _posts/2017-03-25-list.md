## 链式实现
实现一个带头节点的链表






### list.h

```cpp
//#pragma once
//mylist.h
#ifndef _MYLIST_H_
#define  ElymentType int
struct Node;
typedef struct Node* PtrToNode;    //指向节点的指针类型
typedef PtrToNode List;             
typedef PtrToNode Position;

List MakeEmpty(List);
int IsEmpty(List);
int IsLast(Position,List);
Position Find(ElymentType,List);
Position FindPrevious(ElymentType, List);
void Delete(ElymentType, List);
void Delete2(ElymentType, List*);  //使用双重指针实现删除
void Insert(ElymentType, List, Position);
Position Header(List);
Position Fisrt(List);
Position Advance(Position);
void Print(List);
ElymentType Retrieve(Position);

#endif

struct Node
{
	ElymentType Element;
	Position Next;
};
```

### list.c

```cpp
//************************************
// Method:    IsEmpty
// FullName:  IsEmpty
// Access:    public 
// Returns:   int
// Qualifier:
// Parameter: List L
// Note:      判断L是否为空
//************************************
int IsEmpty(List L) {
	return L->Next == NULL;
}

//************************************
// Method:    IsLast
// FullName:  IsLast
// Access:    public 
// Returns:   int
// Qualifier:
// Parameter: Position P
// Parameter: List L
// Note:      判断节点P是否是L的最后一个节点
//************************************
int IsLast(Position P, List L) {
	return P->Next == NULL;
}

//************************************
// Method:    Find
// FullName:  Find
// Access:    public 
// Returns:   Position
// Qualifier:
// Parameter: ElymentType X
// Parameter: List L
// Note:      在L中查找节点P
//************************************
Position Find(ElymentType X, List L) {
	Position P = L->Next;
	while (P != NULL && P->Element != X)
		P = P->Next;
	return P;
}


//************************************
// Method:    FindPrevious
// FullName:  FindPrevious
// Access:    public 
// Returns:   Position
// Qualifier:
// Parameter: ElymentType X
// Parameter: List L
// Note:      在链表L中查找元素X的前面一个节点
//************************************
Position FindPrevious(ElymentType X, List L) {
	Position P = L;
	if (!Find(X, L))
		return NULL;
	while (P->Next->Element != X && P->Next != NULL)
		P = P->Next;
	return P;
}

//************************************
// Method:    Delete
// FullName:  Delete
// Access:    public 
// Returns:   void
// Qualifier:
// Parameter: ElymentType X
// Parameter: List L
// Note:      删除链表中元素值为X的节点
//************************************
void Delete(ElymentType X, List L) {
	Position P, TmpCell;
	P = FindPrevious(X, L);
	if (P != NULL)
	{
		TmpCell = P->Next;
		P->Next = P->Next->Next;
		free(TmpCell);
	}
}

//************************************
// Method:    Delete2
// FullName:  Delete2
// Access:    public 
// Returns:   void
// Qualifier:
// Parameter: ElymentType X
// Parameter: List * curr
// Note:      使用多重指针实现链表的删除操作
//************************************
void Delete2(ElymentType X, List* curr) {
	curr = &(*curr)->Next;
	while (*curr != NULL && (*curr)->Element != X)
		curr = &(*curr)->Next;
	if (*curr != NULL)
		*curr = (*curr)->Next;
}

//************************************
// Method:    Insert
// FullName:  Insert
// Access:    public 
// Returns:   void
// Qualifier:
// Parameter: ElymentType X
// Parameter: List L
// Parameter: Position P
// Note:      在链表L中的P节点之后插入新节点，节点包含的元素值为X
//************************************
void Insert(ElymentType X, List L, Position P) {
	Position TmpCell;
	TmpCell = (Position)malloc(sizeof(Node));
	if (TmpCell == NULL)
		std::cout << "out of space" << endl;
	TmpCell->Element = X;
	TmpCell->Next = P->Next;
	P->Next = TmpCell;
}
```
